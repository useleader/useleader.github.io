```bash
git clone git://sourceware.org/git/binutils-gdb.git

# easy mode: 
docker run -it -v $(pwd):/src aflplusplus/aflplusplus
# better mode:
docker run -it --privileged \
  -v $(pwd)/binutils-gdb:/src \
  -v $(pwd)/aflpp_inputs/binutils-gdb:/inputs \
  -v $(pwd)/aflpp_outputs/binutils-gdb:/outputs \
  aflplusplus/aflplusplus:latest \
  /bin/bash
  
# 或者使用Dockerfile直接制定，这样也避免了clean不彻底带来的环境混乱

```

两种编译插桩方式：gcc、llvm
尝试使用llvm_mode来做fuzz

```bash
# 在容器内验证 AFL++ 工具
afl-clang-fast --version
afl-fuzz --help

# 设置性能优化参数
echo core >/proc/sys/kernel/core_pattern
echo performance | tee /sys/devices/system/cpu/cpu*/cpufreq/scaling_governor

# 检查系统状态
afl-system-config
```

```bash
cd /src

# 彻底清理
make distclean 2>/dev/null || true
git clean -fdx 2>/dev/null || true

# 使用 LLVM 模式的最佳配置
export CC=afl-clang-fast
export CXX=afl-clang-fast++
export AFL_USE_ASAN=1        # 建议开启地址检测（调试时）
export AFL_LLVM_INSTRUMENT=PCGUARD  # 插桩方式（默认即可）

# 配置项目
./configure \
    --disable-shared \
    --disable-nls \
    --disable-werror
```
