# 创新点1：检查LLM生成的语法

语法生成的源代码位于`afl-fuzz.c`中的`setup_llm_grammars`函数中

```c++
void setup_llm_grammars()

{

  

  ACTF("Getting grammars from LLM...");  // 打印一条状态信息，ACTF是afl的日志宏

  

  khash_t(consistency_table) *const_table = kh_init(consistency_table);  // 初始化一个khash（哈希表）指针，用于保存“header -> field_table” 的映射（后面会把每个 message header 对应的字段频次表放这里）。

  char *first_question;

  char *templates_prompt = construct_prompt_for_templates(protocol_name, &first_question);  // 构造发送给LLM的初始prompt，返回templates_prompt，通过first_question输出与后续对话相关的第一个问题

  

  for (int iter = 0; iter < TEMPLATE_CONSISTENCY_COUNT; iter++)  // 每次向LLM请求模板并交叉验证、合并，提高模板的一致性或覆盖性

  {

    klist_t(gram) *grammar_list = kl_init(gram);  // 创建一个klib列表，用来保存LLM解析后的grammars，看着像是链表似的

  

    char *templates_answer = chat_with_llm(templates_prompt, "turbo", GRAMMAR_RETRIES, 0.5); // 使用 chat_with_llm（封装与 LLM 通信的函数）向模型 "turbo" 发送 templates_prompt，重试次数为 GRAMMAR_RETRIES，温度（或相似）为 0.5。

    if (templates_answer == NULL)

      goto free_templates_answer;

  

    printf("## Answer from LLM:\n %s\n", templates_answer); // 查看一下运行的输出内容

    char *remaining_prompt = construct_prompt_for_remaining_templates(protocol_name, first_question, templates_answer); // 基于first_question与templates_answer构造后续prompt，询问LLM以获取后续模板

    printf("remaining prompt is:\n %s\n", remaining_prompt); // 查看一下运行的输出内容

    char *remaining_templates = chat_with_llm(remaining_prompt, "turbo", GRAMMAR_RETRIES, 0.5);

    if (remaining_templates == NULL)

      goto free_remaining;

  

    // printf("## Remaining templates:\n %s\n", remaining_templates);

  

    char *combined_templates = NULL;

    asprintf(&combined_templates, "%s\n%s", templates_answer, remaining_templates); // 把templates, remaining这两个合并为一个字符串

    // TODO: 风险：asprintf 可能返回 -1（内存分配失败）。代码没有检查返回值 — 这是可改进点（应检查返回值并在失败时处理）

    char *grammar_output_path = alloc_printf("%s/protocol-grammars/llm-grammar-output-%d", out_dir, iter); // 保存语法输出

    int grammar_output_fd = open(grammar_output_path, O_WRONLY | O_CREAT, 0600); // O_WRONLY write only O_CREAT if not then create 0600 权限

  

    ck_write(grammar_output_fd, combined_templates, strlen(combined_templates), grammar_output_path); //把指定文件写在指定路径

    close(grammar_output_fd);

    ck_free(grammar_output_path);

  

    extract_message_grammars(combined_templates, grammar_list); // 解析combined_template内容，解析结果输出到grammar_list

  

    kliter_t(gram) * iter;

    for (iter = kl_begin(grammar_list); iter != kl_end(grammar_list); iter = kl_next(iter)) //这里就是和json格式高度相关的部分了，先看extract

    {

      json_object *jobj = kl_val(iter);

  

      json_object *header = json_object_array_get_idx(jobj, 0);

  

      int absent;

  

      const char *header_str = json_object_get_string(header);

  

      khiter_t k = kh_put(consistency_table, const_table, header_str, &absent);

      if (absent)

      {

        khash_t(field_table) *field_table = kh_init(field_table);

        kh_value(const_table, k) = field_table;

      }

  

      for (int i = 1; i < json_object_array_length(jobj); i++)

      {

        const char *v = json_object_get_string(json_object_array_get_idx(jobj, i));

        khash_t(field_table) *field_table = kh_value(const_table, k);

        khiter_t field_k = kh_put(field_table, field_table, v, &absent);

        if (absent)

        {

          kh_value(field_table, field_k) = 0;

        }

        kh_value(field_table, field_k)++;

      }

    }

    kl_destroy_gram(grammar_list);

  

    free(combined_templates);

    free(remaining_templates);

  

  free_remaining:

    free(remaining_prompt);

  

  free_templates_answer:

    free(templates_answer);

  }

  

  int pattern_index = 0;

  for (khiter_t con_t_iter = kh_begin(const_table); con_t_iter != kh_end(const_table); ++con_t_iter)

  {

    if (kh_exist(const_table, con_t_iter))

    {

      pcre2_code **patterns = ck_alloc(2 * sizeof(pcre2_code *));

  

      khash_t(field_table) *field_table = kh_value(const_table, con_t_iter);

  

      json_object *header_v = json_object_new_string(kh_key(const_table, con_t_iter));

      const char *header_str = json_object_to_json_string(header_v);

  

      char *pattern_path = alloc_printf("%s/protocol-grammars/pattern-%d", out_dir, pattern_index);

      pattern_index++;

      int pattern_fd = open(pattern_path, O_WRONLY | O_CREAT, 0600);

  

      char *message_type = extract_message_pattern(header_str, field_table, patterns, pattern_fd, pattern_path);

      if (message_type != NULL)

      {

        int discard;

        kh_put(strSet, message_types_set, message_type, &discard);

        *kl_pushp(rang, protocol_patterns) = patterns;

      }

  

      json_object_put(header_v);

      close(pattern_fd);

      ck_free(pattern_path);

  

    }

  }

  

  free(first_question);

  free(templates_prompt);
}
```
实现的操作：
1. 利用`construct_prompt_for_templates`生成初始的template_prompt与first_answer
2. 通过`chat_with_llm`生成template_answer
3. 利用`construct_prompt_for_remaining_templates`来获取后续的`remaining_prompt`
4. 通过`chat_with_llm`生成remaing_templates
5. 然后根据`extract_message_grammars`进行提炼

辅助函数位于`chat-llm.c`中
`chat_with_llm`函数就实现了一个访问+提取内容的操作
`construct_prompt_for_templates`仿照rtsp, http的例子生成给定协议名的prompt;
```
prompt_grammars = [

        {"role": "system", "content": "You are a helpful assistant."},

        {"role": "user", "content": msg}

    ]
```

construct_prompt_for_remaining_templates直接构造了second_question，然后作为remaining_templates等待llm上下文补全

extract_message_grammars在 answers 文本中按 `[` 到 `]` 的区间粗提取 JSON 片段，json_tokener_parse 成对象后 push 到 grammar_list
# 创新点2：检查增强种子

种子增强的源代码位于`afl-fuzz.c` 的`get_seeds_with_messsage_types` 函数中



1. 遍历目录来读文件
2. 跳过`.`,`..`,`enriched`等文件
3. 将文件读入缓冲区
4. 调用extract_requests （协议特定回调）提取消息区域列表 regions，识别region的消息类型
5. 复制消息类型集合 `message_types_set` 得到一个可删集合 `messages`，然后从其中移除在当前种子中观察到的消息类型（即文件已包含的 header）
6. 若集合清空 → 该种子已经包含所有消息类型，跳过富集。
7. 若集合过大，随机删减到 `MAX_ENRICHMENT_CORPUS_SIZE` 以内。
8. 将剩余message集合生成所有组合子集，大小为 `MAX_ENRICHMENT_MESSAGE_TYPES`
	1. 对每个子集使用Enrich_sequence函数来实现llm填充
	2. 判断边界条件与格式化文件
9. 释放资源继续执行

辅助函数位于 `chat-llm.c` 中

enrich_sequence函数

- 输出提示信息
    
- 调用get_seeds_with_messsage_types(in_dir, message_types_set)，这一步会将种子保存到in_dir中，扩充的数量还是挺多的，只有框起来的部分是初始种子。
